    <script>
        // scripts/cadastro.js (O código que você me enviou, com LOGS adicionais)
        console.log("-> cadastro.js starting execution (before DOMContentLoaded)."); // DEBUG

        document.addEventListener('DOMContentLoaded', function() {
            console.log("DOMContentLoaded fired in cadastro.js."); // DEBUG

            const registerForm = document.getElementById('registerForm');
            const firstName = document.getElementById('firstName');
            const lastName = document.getElementById('lastName');
            const email = document.getElementById('email');
            const password = document.getElementById('password');
            const birthday = document.getElementById('birthday');
            const gender = document.getElementById('gender');
            const terms = document.getElementById('terms');

            if (!registerForm) { // Explicit check
                console.error("ERROR: 'registerForm' (ID) not found in the HTML! Submit listener NOT attached."); // CRITICAL DEBUG
                return; // Stop execution if form is not found
            } else {
                console.log("'registerForm' element found."); // DEBUG
            }

            registerForm.addEventListener('submit', async function(e) {
                console.log("Submit event fired on registerForm."); // DEBUG
                e.preventDefault();
                
                console.log("Calling validateForm()..."); // DEBUG
                if (!validateForm()) {
                    console.log("validateForm() returned false. Registration aborted."); // DEBUG
                    return;
                }
                console.log("validateForm() returned true. Proceeding with Firebase registration..."); // DEBUG
                
                showLoadingState();
                console.log("showLoadingState() called. Attempting Firebase registration..."); // DEBUG
                
                try {
                    const userCredential = await window.firebaseAuth.createUserWithEmailAndPassword(email.value.trim(), password.value);
                    console.log('User successfully registered in Firebase:', userCredential.user); // DEBUG

                    hideLoadingState();
                    alert('¡Cuenta creada exitosamente! Redirecionando para o login.');
                    window.location.href = 'index.html';
                    
                } catch (error) {
                    hideLoadingState();
                    console.error('Firebase registration error (catch block):', error.code, error.message); // DEBUG

                    let friendlyErrorMessage = 'Erro ao criar conta. Por favor, tente novamente.';
                    if (error.code === 'auth/email-already-in-use') {
                        friendlyErrorMessage = 'Este e-mail já está cadastrado. Tente fazer login ou use outro e-mail.';
                        showFieldError(email, friendlyErrorMessage);
                    } else if (error.code === 'auth/invalid-email') {
                        friendlyErrorMessage = 'O formato do e-mail é inválido.';
                        showFieldError(email, friendlyErrorMessage);
                    } else if (error.code === 'auth/weak-password') {
                        friendlyErrorMessage = 'A senha é muito fraca. Use pelo menos 6 caracteres e combine letras, números e símbolos.';
                        showFieldError(password, friendlyErrorMessage);
                    } else if (error.code === 'auth/operation-not-allowed') {
                        friendlyErrorMessage = 'O registro de e-mail/senha não está ativado. Entre em contato com o suporte.';
                    } else {
                         friendlyErrorMessage += ` (Detalhe: ${error.message})`; // Fallback para outros erros
                    }

                } finally {
                    hideLoadingState();
                }

            function validateForm() {
                console.log("Executing validateForm()..."); // DEBUG
                let isValid = true;
                
                if (firstName.value.trim().length < 2) {
                    showFieldError(firstName, 'El nombre debe tener al menos 2 caracteres');
                    isValid = false;
                } else {
                    clearFieldError(firstName);
                }
                
                const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
                if (!emailRegex.test(email.value.trim())) {
                    showFieldError(email, 'Por favor, ingrese un email válido');
                    isValid = false;
                } else {
                    clearFieldError(email);
                }

                if (password.value.length < 6) {
                    showFieldError(password, 'La contraseña debe tener al menos 6 caracteres');
                    isValid = false;
                } else {
                    clearFieldError(password);
                }
                
                if (!terms.checked) {
                    showFieldError(terms, 'Debe aceptar los términos y condiciones');
                    isValid = false;
                } else {
                    clearFieldError(terms);
                }
                
                console.log("validateForm() finished, returning:", isValid); // DEBUG
                return isValid;
            }

            function showFieldError(field, message) {
                clearFieldError(field);
                field.style.borderColor = '#e74c3c';
                const errorDiv = document.createElement('div');
                errorDiv.className = 'field-error';
                errorDiv.textContent = message;
                errorDiv.style.color = '#e74c3c';
                errorDiv.style.fontSize = '12px';
                errorDiv.style.marginTop = '5px';
                if (field.type === 'checkbox') {
                    field.parentNode.parentNode.appendChild(errorDiv);
                } else {
                    field.parentNode.appendChild(errorDiv);
                }
            }

            function clearFieldError(field) {
                field.style.borderColor = '#e1e1e1';
                const parent = field.type === 'checkbox' ? field.parentNode.parentNode : field.parentNode;
                const existingError = parent.querySelector('.field-error');
                if (existingError) {
                    existingError.remove();
                }
            }

            function showLoadingState() {
                const submitBtn = registerForm.querySelector('button[type="submit"]');
                if (submitBtn) {
                    submitBtn.disabled = true;
                    submitBtn.textContent = 'Criando conta...';
                    submitBtn.style.opacity = '0.7';
                }
            }

            function hideLoadingState() {
                const submitBtn = registerForm.querySelector('button[type="submit"]');
                if (submitBtn) {
                    submitBtn.disabled = false;
                    submitBtn.textContent = 'Cadastrar';
                    submitBtn.style.opacity = '1';
                }
            }

            firstName.addEventListener('blur', function() {

                } else {
                    const birthDate = new Date(this.value);
                    const today = new Date();
                    const age = today.getFullYear() - birthDate.getFullYear();
                    const monthDiff = today.getMonth() - birthDate.getMonth();
                    
                    if (age < 13 || (age === 13 && monthDiff < 0)) {
                        showFieldError(this, 'Debe tener al menos 13 años para registrarse');
                    } else {
                        clearFieldError(this);
                    }
                }
